# estimate advantages using GAE generalized advantage estimation
def advantages_est(values, masks, rewards):
    returns = []
    gae = 0
    # reverse because it's a iterative formula from t+1 -> t
    for i in reversed(range(len(rewards))):
      # discounted rewards
        # mask means: if game is over, we don't want this value in GAE 
        delta = rewards[i] + gamma * values[i + 1] * masks[i] - values[i]
        # lmbda is the smoothing parameter
        gae = delta + gamma * lmbda * masks[i] * gae
        # reverse again
        returns.insert(0, gae + values[i])

    adv = np.array(returns) - values[:-1]
    # standardize according to normal dist
    adv_st = (adv - np.mean(adv)) / (np.std(adv) + 1e-10)
    return returns, adv_st

def ppo_loss(oldpolicy_probs, advantages, rewards, values):
    def loss(y_true, y_pred):
        
        newpolicy_probs = y_pred
        
        # new policy / new policy = exp(log(new policy) - log(old policy)) (add 1e-8 to avoid blow-up)
        ratio = K.exp(K.log(newpolicy_probs + 1e-8) - K.log(oldpolicy_probs + 1e-8))
        
        # actor loss as min(ratio x advantages, (epsilon-clip of ratio) x advantage)
        # (just following the formula from paper)
        val_1 = ratio * advantages
        val_2 = K.clip(ratio, min_value=1 - clipping_val, max_value=1 + clipping_val) * advantages
        actor_loss = -K.mean(K.minimum(val_1, val_2))
        
        critic_loss = K.mean(K.square(rewards - values))
        
        total_loss = critic_discount * critic_loss + actor_loss - entropy_beta * K.mean(
            -(newpolicy_probs * K.log(newpolicy_probs + 1e-8)))
        return total_loss

    return loss


def get_actor(input_dims, output_dims):
  # defines the actor model (net)
  # uses ppo_loss defined above
    state_input = Input(shape=input_dims)
    oldpolicy_probs = Input(shape=(1, output_dims,))
    advantages = Input(shape=(1, 1,))
    rewards = Input(shape=(1, 1,))
    values = Input(shape=(1, 1,))

    # F-C net (using Keras)
    x = Dense(512, activation='relu', name='fc1')(state_input)
    x = Dense(256, activation='relu', name='fc2')(x)
    # use softmax to get out probabilities
    out_actions = Dense(n_actions, activation='softmax', name='predictions')(x)

    model = Model(inputs=[state_input, oldpolicy_probs, advantages, rewards, values],
                  outputs=[out_actions])
    model.compile(optimizer=Adam(lr=1e-4), loss=[ppo_loss(
        oldpolicy_probs=oldpolicy_probs,
        advantages=advantages,
        rewards=rewards,
        values=values)])
    # model.summary()
    return model

def get_critic(input_dims):
  # net for critic, loss here is just mse
    state_input = Input(shape=input_dims)

    # F-C net (using Keras)
    x = Dense(512, activation='relu', name='fc1')(state_input)
    x = Dense(256, activation='relu', name='fc2')(x)
    # final activation is tanh
    out_actions = Dense(1, activation='tanh')(x)

    model = Model(inputs=[state_input], outputs=[out_actions])
    model.compile(optimizer=Adam(lr=1e-4), loss='mse')
    # model.summary()
    return model
